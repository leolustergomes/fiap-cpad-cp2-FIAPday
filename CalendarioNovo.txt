import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  TouchableOpacity,
  StyleSheet,
  TextInput,
  ScrollView,
} from "react-native";

import AsyncStorage from "@react-native-async-storage/async-storage";
import * as Notifications from "expo-notifications";

// 🔔 Configuração básica de comportamento da notificação
Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: true,
    shouldSetBadge: false,
  }),
});

export default function App({ navigation }: any) {
  const [texto, setTexto] = useState("");
  const [eventos, setEventos] = useState<any>({});
  const [diaSelecionado, setDiaSelecionado] = useState<number | null>(null);

  // 🔐 Permissão de notificação
  useEffect(() => {
    async function pedirPermissao() {
      const { status } = await Notifications.requestPermissionsAsync();
      if (status !== "granted") {
        alert("Permissão de notificação negada");
      }
    }

    pedirPermissao();
  }, []);

  // 🔔 Função para agendar notificação
  async function agendarNotificacao(texto: string, dia: number) {
    await Notifications.scheduleNotificationAsync({
      content: {
        title: "📅 Lembrete de Evento",
        body: `${texto} (Dia ${dia})`,
      },
      trigger: {
        seconds: 10, // ⏱️ tempo do lembrete (pode mudar depois)
      },
    });
  }

  function salvarEvento() {
    if (!texto.trim() || !diaSelecionado) return;

    setEventos((prev: any) => ({
      ...prev,
      [diaSelecionado]: [...(prev[diaSelecionado] || []), texto],
    }));

    // 🔥 dispara notificação
    agendarNotificacao(texto, diaSelecionado);

    setTexto("");
  }

  const dias = Array.from({ length: 31 }, (_, i) => i + 1);
  const lista = eventos[diaSelecionado as number] || [];

  return (
    <ScrollView style={styles.container}>
      {/* 🔴 LOGOUT */}
      <TouchableOpacity
        style={styles.botao}
        onPress={async () => {
          await AsyncStorage.removeItem("logado");
          navigation.navigate("Login");
        }}
      >
        <Text style={styles.textoBotao}>Logout</Text>
      </TouchableOpacity>

      <Text style={styles.titulo}>📅 Março 2026 🚀</Text>

      <View style={styles.grade}>
        {dias.map((dia) => (
          <TouchableOpacity
            key={dia}
            style={[
              styles.dia,
              diaSelecionado === dia && styles.diaSelecionado,
            ]}
            onPress={() => setDiaSelecionado(dia)}
          >
            <Text style={styles.textoDia}>{dia}</Text>
          </TouchableOpacity>
        ))}
      </View>

      <Text style={styles.info}>
        {diaSelecionado
          ? `Dia selecionado: ${diaSelecionado}`
          : "Selecione um dia"}
      </Text>

      <TextInput
        style={styles.input}
        placeholder="Digite um evento"
        placeholderTextColor="#999"
        value={texto}
        onChangeText={setTexto}
      />

      <TouchableOpacity style={styles.botao} onPress={salvarEvento}>
        <Text style={styles.textoBotao}>Salvar</Text>
      </TouchableOpacity>

      {lista.length === 0 ? (
        <Text style={styles.semEvento}>Nenhum evento para este dia</Text>
      ) : (
        lista.map((item: string, index: number) => (
          <Text key={index} style={styles.evento}>
            • {item}
          </Text>
        ))
      )}
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    paddingTop: 50,
    paddingHorizontal: 15,
    backgroundColor: "#000",
  },

  titulo: {
    fontSize: 26,
    textAlign: "center",
    marginBottom: 20,
    fontWeight: "bold",
    color: "#ff4d4d",
  },

  grade: {
    flexDirection: "row",
    flexWrap: "wrap",
    marginBottom: 20,
  },

  dia: {
    width: "14.2%",
    aspectRatio: 1,
    backgroundColor: "#111",
    justifyContent: "center",
    alignItems: "center",
    margin: 2,
    borderRadius: 8,
    borderWidth: 1,
    borderColor: "#ff4d4d",
  },

  diaSelecionado: {
    backgroundColor: "#ff4d4d",
  },

  textoDia: {
    color: "#fff",
    fontWeight: "bold",
  },

  info: {
    color: "#fff",
    marginBottom: 10,
    textAlign: "center",
  },

  input: {
    backgroundColor: "#111",
    padding: 12,
    borderRadius: 8,
    marginBottom: 10,
    color: "#fff",
    borderWidth: 1,
    borderColor: "#ff4d4d",
  },

  botao: {
    backgroundColor: "#ff4d4d",
    padding: 12,
    alignItems: "center",
    borderRadius: 8,
    marginBottom: 15,
  },

  textoBotao: {
    color: "#fff",
    fontWeight: "bold",
  },

  evento: {
    fontSize: 16,
    marginTop: 5,
    color: "#fff",
  },

  semEvento: {
    textAlign: "center",
    marginTop: 10,
    color: "#aaa",
  },
});
